function doPost(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheets()[0]; var data = sheet.getDataRange().getValues(); var headers = data[0]; var action = (e.parameter.action || "").trim(); if (action === "saveExplanation") { var id = (e.parameter.Id || "").trim(); var term = (e.parameter.Term || "").trim(); var explanation = (e.parameter.Explanation || "").trim(); var idCol = headers.indexOf("Id"); var termCol = headers.indexOf("Term"); var explanationCol = headers.indexOf("UserExplanation"); if (idCol === -1 || termCol === -1 || explanationCol === -1) { return ContentService .createTextOutput(JSON.stringify({ success: false, error: "Missing required columns: Id, Term, or UserExplanation" })) .setMimeType(ContentService.MimeType.JSON); } for (var i = 1; i < data.length; i++) { var rowId = String(data[i][idCol] || "").trim(); var rowTerm = String(data[i][termCol] || "").trim(); if (rowId === id && rowTerm === term) { sheet.getRange(i + 1, explanationCol + 1).setValue(explanation); return ContentService .createTextOutput(JSON.stringify({ success: true, message: "Explanation saved", row: i + 1 })) .setMimeType(ContentService.MimeType.JSON); } } return ContentService .createTextOutput(JSON.stringify({ success: false, error: "Matching row not found" })) .setMimeType(ContentService.MimeType.JSON); } var timestamp = new Date(); var userInput = (e.parameter.UserInput || "").trim(); var tag = (e.parameter.Tag || "").trim(); var front = (e.parameter.Front || "").trim(); var back = (e.parameter.Back || "").trim(); sheet.appendRow([timestamp, userInput, tag, front, back]); return ContentService .createTextOutput(JSON.stringify({ success: true, message: "Success" })) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheets()[0]; var data = sheet.getDataRange().getValues(); var headers = data[0]; var setId = ((e.parameter && e.parameter.set) || "").trim(); var idCol = headers.indexOf("Id"); var tagCol = headers.indexOf("Tag"); var termCol = headers.indexOf("Term"); var definitionCol = headers.indexOf("Definition"); if (idCol === -1 || tagCol === -1 || termCol === -1 || definitionCol === -1) { return ContentService .createTextOutput(JSON.stringify({ success: false, error: "Missing required columns: Id, Tag, Term, or Definition" })) .setMimeType(ContentService.MimeType.JSON); } var result = []; for (var i = 1; i < data.length; i++) { var row = data[i]; var id = String(row[idCol] || "").trim(); var tag = String(row[tagCol] || "").trim(); var term = String(row[termCol] || "").trim(); var definition = String(row[definitionCol] || "").trim(); if (tag === setId && term && definition) { result.push({ id: id, front: term, back: definition }); } } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); }
